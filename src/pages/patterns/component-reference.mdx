export const COMPONENT_REFERENCE_QUERY = graphql`
  query {
    allComponentMetadata(sort: {fields: displayName, order: ASC}) {
      nodes {
        displayName
        description {
          childMdx {
            body
          }
        }
        childrenComponentProp {
          name
          docblock
          required
          parentType {
            name
          }
          type {
            value
          }
          defaultValue {
            value
            computed
          }
        }
        parent {
          ... on File {
            relativePath
          }
        }
      }
    }
  }
`
